# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
# LOOKUP вҖ” busca tour por origin_key no banco
# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
def find_tour_by_origin_key(origin_key: str, auth: tuple) -> int | None:
    """Retorna WP post ID ou None."""
    r = requests.get(
        f"{BASE}/wp/v2/vana_tour",
        params={
            "per_page": 100,
            "status":   "any",
            "_fields":  "id,slug,meta",
        },
        auth=auth,
        timeout=15,
    )
    r.raise_for_status()
    tours = r.json()

    for t in tours:
        meta = t.get("meta", {})
        if meta.get("_vana_origin_key") == origin_key:
            return t["id"]

    # Fallback: busca via wp-json direto no postmeta
    return None


# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
# BUILD вҖ” monta body WP REST
# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
def build_wp_body(tour: dict) -> dict:
    """Converte tour-sample.json para body da WP REST API."""
    itinerary_json = json.dumps(
        tour.get("itinerary", []),
        ensure_ascii=False
    )

    return {
        "title":       tour.get("title", ""),
        "status":      "publish",
        "slug":        tour.get("slug", ""),
        "meta": {
            "_vana_origin_key":   tour.get("origin_key", ""),
            "_vana_start_date":   tour.get("start_date", ""),
            "_tour_subtitle":     tour.get("subtitle", ""),
            "_tour_theme":        tour.get("theme", ""),
            "_tour_description":  tour.get("description", ""),
            "_tour_dates":        tour.get("dates", ""),
            "_tour_organizers":   tour.get("organizers", ""),
            "_tour_is_current":   "1" if tour.get("is_current") else "",
            "_tour_itinerary":    itinerary_json,
        },
    }


# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
# UPSERT
# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
def upsert_tour(tour: dict, auth: tuple, dry_run: bool = False) -> dict:
    origin_key = tour.get("origin_key", "")
    body       = build_wp_body(tour)

    _banner(tour, origin_key)

    if dry_run:
        print("\nрҹ§Ә DRY-RUN вҖ” body que seria enviado:\n")
        print(json.dumps(body, ensure_ascii=False, indent=2))
        return {"dry_run": True}

    # Tenta encontrar tour existente pelo origin_key
    existing_id = None
    try:
        existing_id = find_tour_by_origin_key(origin_key, auth)
    except Exception as e:
        print(f"   вҡ пёҸ  Lookup falhou ({e}) вҖ” tentarГЎ criar novo.")

    if existing_id:
        print(f"\nрҹ”„ Tour existente encontrado (ID {existing_id}) вҖ” atualizando...")
        r = requests.post(
            f"{BASE}/wp/v2/vana_tour/{existing_id}",
            json=body, auth=auth, timeout=20,
        )
        action = "updated"
    else:
        print("\nвңЁ Tour novo вҖ” criando...")
        r = requests.post(
            f"{BASE}/wp/v2/vana_tour",
            json=body, auth=auth, timeout=20,
        )
        action = "created"

    print(f"   HTTP {r.status_code}")

    if not r.ok:
        print(f"   вқҢ {r.text[:400]}")
        return {"error": r.text}

    result = r.json()
    _print_success(result, action)
    return result


def _banner(tour: dict, origin_key: str):
    print(f"\n{'в•җ'*60}")
    print(f"  рҹ“Ў {BASE}/wp/v2/vana_tour")
    print(f"  рҹ”‘ origin_key  : {origin_key}")
    print(f"  рҹ“қ title       : {tour.get('title','вҖ”')}")
    print(f"  рҹ“… start_date  : {tour.get('start_date','вҖ”')}")
    print(f"  рҹҢҚ theme       : {tour.get('theme','вҖ”')}")
    print(f"  вӯҗ is_current  : {tour.get('is_current', False)}")
    print(f"{'в•җ'*60}")


def _print_success(result: dict, action: str):
    icons = {"created": "вң… CRIADO", "updated": "рҹ”„ ATUALIZADO"}
    print(f"\n  {icons.get(action,'вң…')}")
    print(f"  tour_id   : {result.get('id','вҖ”')}")
    print(f"  slug      : {result.get('slug','вҖ”')}")
    print(f"  status    : {result.get('status','вҖ”')}")
    if result.get("link"):
        print(f"  link      : {result['link']}")


# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
# VALIDAГҮГғO
# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
def validate(tour: dict) -> list[str]:
    errors = []
    for f in ["origin_key", "title", "start_date"]:
        if not tour.get(f):
            errors.append(f"Campo obrigatГіrio ausente: '{f}'")
    ok = tour.get("origin_key", "")
    if ok and not ok.startswith("tour:"):
        errors.append(f"'origin_key' deve comeГ§ar com 'tour:' вҖ” tem: '{ok}'")
    sd = tour.get("start_date", "")
    if sd:
        try:
            datetime.strptime(sd, "%Y-%m-%d")
        except ValueError:
            errors.append(f"'start_date' deve ser YYYY-MM-DD вҖ” tem: '{sd}'")
    return errors


# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
# MAIN
# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
def main():
    parser = argparse.ArgumentParser(
        description="Ingest de tour вҖ” wp/v2/vana_tour (Application Password)"
    )
    parser.add_argument(
        "json_file", nargs="?",
        help="JSON do tour. Omita para usar o TOUR_SCHEMA embutido."
    )
    parser.add_argument("--dry-run",         action="store_true")
    parser.add_argument("--skip-validation", action="store_true")
    args = parser.parse_args()

    # 1. Carrega payload
    if args.json_file:
        path = Path(args.json_file)
        if not path.exists():
            print(f"вқҢ Arquivo nГЈo encontrado: {path}")
            sys.exit(1)
        with open(path, "r", encoding="utf-8") as f:
            try:
                tour = json.load(f)

# HMAC вҖ” espelha exatamente o class-vana-hmac.php
# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
def _sign(body_str: str) -> dict:
    timestamp = str(int(time.time()))
    nonce     = secrets.token_hex(16)
    message   = f"{timestamp}\n{nonce}\n{body_str}"
    signature = hmac.new(
        INGEST_SECRET.encode("utf-8"),
        message.encode("utf-8"),
        hashlib.sha256,
    ).hexdigest()
    return {
        "vana_timestamp": timestamp,
        "vana_nonce":     nonce,
        "vana_signature": signature,
    }

    if isinstance(obj, list):
        return [_clean(i) for i in obj]
    return obj

# VALIDAГҮГғO do schema v2.6 / 3.1

VALID_STATUSES = {"done", "live", "upcoming", ""}

def validate(payload: dict) -> list[str]:
    errors = []

    # Raiz
    for field in ["visit_id", "title_pt", "timezone", "days"]:
        if not payload.get(field):
            errors.append(f"Raiz: campo obrigatГіrio ausente вҶ’ '{field}'")

    days = payload.get("days", [])
    if not isinstance(days, list) or len(days) == 0:
        errors.append("'days' deve ser uma lista nГЈo-vazia.")
        return errors
    if len(days) > 400:
        errors.append(f"'days' excede 400 itens (tem {len(days)}).")

    for i, day in enumerate(days):
        label = day.get("date_local") or f"days[{i}]"

        # Campos obrigatГіrios do dia
        if not day.get("date_local"):
            errors.append(f"{label}: 'date_local' ausente.")
        if not day.get("label_pt"):
            errors.append(f"{label}: 'label_pt' ausente.")

        # Hero
        hero = day.get("hero")
        if not hero:
            errors.append(f"{label}: 'hero' ausente.")
        else:
            if not hero.get("title_pt"):
                errors.append(f"{label} > hero: 'title_pt' ausente.")
            media_fields = ["youtube_url", "instagram_url", "facebook_url", "drive_url"]
            if not any(hero.get(f) for f in media_fields):
                errors.append(f"{label} > hero: nenhuma fonte de mГӯdia (youtube/instagram/facebook/drive).")

        # Schedule
        for j, s in enumerate(day.get("schedule", [])):
            ref = f"{label} > schedule[{j}]"
            if not s.get("time_local"):
                errors.append(f"{ref}: 'time_local' ausente.")
            if not s.get("title_pt"):
                errors.append(f"{ref}: 'title_pt' ausente.")
            status = s.get("status", "")
            if status not in VALID_STATUSES:
                errors.append(f"{ref}: status invГЎlido '{status}' вҖ” use done/live/upcoming.")

        # VODs
        for j, vod in enumerate(day.get("vods", [])):
            ref = f"{label} > vods[{j}]"
            if not vod.get("title_pt"):
                errors.append(f"{ref}: 'title_pt' ausente.")
            media_fields = ["youtube_url", "facebook_url", "drive_url"]
            if not any(vod.get(f) for f in media_fields):
                errors.append(f"{ref}: nenhuma fonte de mГӯdia.")

        # Photos
        for j, ph in enumerate(day.get("photos", [])):
            ref = f"{label} > photos[{j}]"
            if not ph.get("full_url") and not ph.get("thumb_url"):
                errors.append(f"{ref}: precisa de 'full_url' ou 'thumb_url'.")

        # Sangha moments
        for j, sm in enumerate(day.get("sangha_moments", [])):
            ref = f"{label} > sangha_moments[{j}]"
            if not sm.get("text_pt"):
                errors.append(f"{ref}: 'text_pt' ausente.")
            if not sm.get("author"):
                errors.append(f"{ref}: 'author' ausente.")

    return errors

# SEND
# в•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җв•җ
def send(envelope: dict, dry_run: bool = False) -> dict:
    body_str = json.dumps(envelope, ensure_ascii=False, separators=(",", ":"))
    params   = _sign(body_str)

    _banner(envelope, body_str)

    if dry_run:
        print("\nрҹ§Ә DRY-RUN вҖ” payload NГғO enviado ao WordPress.")
        print("\nрҹ“Ӣ Envelope preview:\n")
        # Preview com data truncada para legibilidade
        preview = {**envelope, "data": {
            **{k: v for k, v in envelope["data"].items() if k != "days"},
            "days": f"[{len(envelope['data'].get('days', []))} dias вҖ” omitido no preview]"
        }}
        print(json.dumps(preview, ensure_ascii=False, indent=2))
        return {"dry_run": True, "body_bytes": len(body_str)}

    print("\nрҹҡҖ Enviando...")
    try:
        resp = requests.post(
            ENDPOINT,
            params=params,
            data=body_str.encode("utf-8"),
            headers={"Content-Type": "application/json"},
            timeout=30,
        )
    except requests.exceptions.ConnectionError:
        print(f"вқҢ NГЈo foi possГӯvel conectar a {ENDPOINT}")
        sys.exit(1)
    except requests.exceptions.Timeout:
        print("вқҢ Timeout apГіs 30s")
        sys.exit(1)

    print(f"   HTTP {resp.status_code}")

    try:
        result = resp.json()
    except Exception:
        result = {"raw": resp.text[:500]}

    if resp.ok:
        _print_success(result)
    else:
        print(f"\nвқҢ Erro HTTP {resp.status_code}")
        print(f"   {resp.text[:500]}")

    return result

    print(f"  рҹ“қ title             : {envelope.get('title','вҖ”')}")
    print(f"  рҹ“Ұ dias              : {len(envelope['data'].get('days', []))}")
    print(f"  рҹ“Ҹ payload           : {len(body_str):,} bytes")
    print(f"  рҹ•җ updated_at        : {envelope['data'].get('updated_at','вҖ”')}")
    print(f"{'в•җ'*62}")

    parser.add_argument("--skip-validation", action="store_true", help="ForГ§a envio sem validar")
    args = parser.parse_args()

    # 1. LГӘ JSON
    json_path = Path(args.json_file)
    if not json_path.exists():
        print(f"вқҢ Arquivo nГЈo encontrado: {json_path}")

    with open(json_path, "r", encoding="utf-8") as f:
        try:
            payload = json.load(f)
        except json.JSONDecodeError as e:
            print(f"вқҢ JSON invГЎlido: {e}")
            sys.exit(1)

    print(f"\nвң… JSON carregado: {json_path.name}")

    print(f"   title_pt : {payload.get('title_pt','вҖ”')}")
    print(f"   dias     : {len(payload.get('days', []))}")

    # 2. Valida schema
    if not args.skip_validation:
        errors = validate(payload)
        if errors:
            print(f"\nвҡ пёҸ  {len(errors)} erro(s) de validaГ§ГЈo:")
            for e in errors:
                print(f"   вҖў {e}")
            print("\n   Use --skip-validation para forГ§ar o envio.")
            sys.exit(1)
        print("   вң… ValidaГ§ГЈo OK")

    # 3. Monta envelope
    try:
        envelope = build_envelope(payload, args.tour_key)
    except ValueError as e:
        print(e)
        sys.exit(1)

    # 4. Envia
    result = send(envelope, dry_run=args.dry_run)

    # 5. Salva log (apenas envio real)
    if not args.dry_run:
        save_log(json_path, payload, args.tour_key, result)

    print(f"вқҢ NГЈo encontrei: {secrets_path}")
    sys.exit(1)

with open(secrets_path, "rb") as f:
    secrets = tomllib.load(f)

WP_URL        = secrets["vana"]["api_base"].replace("/wp-json", "")
WP_USER       = secrets["vana"]["wp_user"]
WP_APP_PASS   = secrets["vana"]["wp_app_password"]
GITHUB_TOKEN  = secrets["github"]["token"]
GITHUB_REPO   = secrets["github"]["repo"]

    "Accept":               "application/vnd.github+json",

# в”Җв”Җ 1. Busca post no WordPress в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
# в”Җв”Җ 1. Busca post no WordPress в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def fetch_from_wp(slug: str) -> dict:
    url = WP_URL + "/wp-json/wp/v2/vana_visit"   # вҶҗ era "posts", agora "vana_visit"
    r = requests.get(
        url,
        params={"slug": slug, "_fields": "id,title,slug,status,meta"},
        auth=(WP_USER, WP_APP_PASS),
        timeout=15,
    )
    r.raise_for_status()
    posts = r.json()
    if not posts:
        raise ValueError(f"Post '{slug}' nГЈo encontrado no WordPress!")
    return posts[0]

# в”Җв”Җ 2. Busca meta completo в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def fetch_visit_meta(post_id: int) -> dict:

    r = requests.get(
        url,
        params={"_fields": "id,slug,title,meta"},
        auth=(WP_USER, WP_APP_PASS),
        timeout=15,
    )
    r.raise_for_status()
    return r.json()


# в”Җв”Җ 2. Busca meta completo в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def fetch_visit_meta(post_id: int) -> dict:
    url = f"{WP_URL}/wp-json/wp/v2/posts/{post_id}"
    r = requests.get(
        url,
        params={"_fields": "id,slug,title,meta"},
        auth=(WP_USER, WP_APP_PASS),
        timeout=15,
    )
    r.raise_for_status()
    return r.json()


# в”Җв”Җ 3. Monta visit.json в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def build_visit_json(post: dict, slug: str) -> dict:
    meta = post.get("meta", {})

    # Tenta extrair JSON completo do meta _vana_data
    vana_data = meta.get("_vana_data") or meta.get("vana_data") or {}
    if isinstance(vana_data, str):
        try:
            vana_data = json.loads(vana_data)
        except Exception:
            vana_data = {}

    if vana_data and vana_data.get("visit_id"):
        print("  вң… Dados completos em _vana_data")
        return vana_data

    # Fallback mГӯnimo
    print("  вҡ пёҸ  Meta vazio вҖ” estrutura mГӯnima (preencha days manualmente)")
    return {
        "visit_id":       slug,
        "title_pt":       post.get("title", {}).get("rendered", slug),
        "title_en":       "",
        "timezone":       "America/Sao_Paulo",
        "schema_version": "3.1",
        "updated_at":     datetime.now(timezone.utc).isoformat(),
        "days":           [],
        "_note":          "Gerado por sync_visit_to_github.py вҖ” preencher days",
    }


# в”Җв”Җ 4. Salva no GitHub в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def save_to_github(visit_ref: str, visit_json: dict) -> bool:
    path    = f"visits/{visit_ref}/visit.json"
    url     = f"{GITHUB_API}/repos/{GITHUB_REPO}/contents/{path}"
    content = json.dumps(visit_json, ensure_ascii=False, indent=2)
    encoded = base64.b64encode(content.encode("utf-8")).decode("utf-8")

    existing = requests.get(
        url, headers=HEADERS_GH,
        params={"ref": GITHUB_BRANCH},
        timeout=10,
    )
    sha = existing.json().get("sha") if existing.status_code == 200 else None

    body = {
        "message": f"[sync] {visit_ref} via sync_visit_to_github.py",
        "content": encoded,
        "branch":  GITHUB_BRANCH,
    }
    if sha:
        body["sha"] = sha
        print(f"  рҹ”„ Atualizando (sha: {sha[:8]}...)")
    else:
        print(f"  вңЁ Criando novo arquivo")

    resp = requests.put(url, headers=HEADERS_GH, json=body, timeout=20)
    if resp.ok:
        print(f"  вң… {resp.json()['content']['html_url']}")
        return True
    else:
        print(f"  вқҢ {resp.status_code}: {resp.text[:300]}")
        return False


# в”Җв”Җ Main в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def main():
    if len(sys.argv) < 2:
        print("Uso: python sync_visit_to_github.py <slug>")
        print("Ex:  python sync_visit_to_github.py sao-paulo-janeiro-2026")
        sys.exit(1)

    slug = sys.argv[1].strip()
    print(f"\n{'в•җ'*55}")
    print(f"  рҹ”„ Sync: {slug}")
    print(f"  WP  вҶ’ {WP_URL}")

# в”Җв”Җ Assinatura в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def _sign_body(body_str: str) -> dict:
    """POST вҖ” assina timestamp + nonce + body."""
    timestamp = str(int(time.time()))
    nonce     = secrets.token_hex(16)
    message   = f"{timestamp}\n{nonce}\n{body_str}"
    signature = hmac.new(                          # вҶҗ BUG no original
        _secret().encode("utf-8"),
        message.encode("utf-8"),
        hashlib.sha256,
    ).hexdigest()
    return {"vana_timestamp": timestamp,
            "vana_nonce":     nonce,
            "vana_signature": signature}


def _sign_empty() -> dict:
    """GET / DELETE вҖ” body vazio."""
    timestamp = str(int(time.time()))
    nonce     = secrets.token_hex(16)
    message   = f"{timestamp}\n{nonce}\n"
    signature = hmac.new(
        _secret().encode("utf-8"),
        message.encode("utf-8"),
        hashlib.sha256,
    ).hexdigest()
    return {"vana_timestamp": timestamp,
            "vana_nonce":     nonce,
            "vana_signature": signature}


# в”Җв”Җ POST genГ©rico в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def _post(endpoint: str, payload: dict) -> dict:
    body_str = json.dumps(payload, ensure_ascii=False, separators=(",", ":"))
    params   = _sign_body(body_str)
    url      = f"{_base()}/vana/v1/{endpoint.lstrip('/')}"

    resp = requests.post(
        url,
        params=params,
        data=body_str.encode("utf-8"),
        headers={"Content-Type": "application/json"},
        timeout=30,
    )
    resp.raise_for_status()
    return resp.json()


# в”Җв”Җ GET autenticado в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
def _get(endpoint: str, extra_params: dict = None) -> dict:
    params = {**_sign_empty(), **(extra_params or {})}
    url    = f"{_base()}/vana/v1/{endpoint.lstrip('/')}"
    resp   = requests.get(url, params=params, timeout=15)
    resp.raise_for_status()
    return resp.json()

def _delete(endpoint: str, force: bool = False) -> dict:
    params = {**_sign_empty(), "force": int(force)}

    url = (
        f"{WP_BASE_URL}/vana_tour"
        f"?per_page={per_page}"
        f"&_fields=id,title,status,slug,meta"
        f"&orderby=date&order=desc"
    )
    resp = requests.get(url, auth=WP_AUTH, timeout=15)
    resp.raise_for_status()

    items = []
    for t in resp.json():
        meta = t.get("meta", {})
        items.append({
            "id":         t.get("id"),
            "title":      t.get("title", {}).get("rendered", f"ID {t.get('id')}"),
            "status":     t.get("status", "вҖ”"),
            "slug":       t.get("slug", "вҖ”"),
            "origin_key": meta.get("_vana_origin_key", "вҖ”"),
            "permalink":  f"https://beta.vanamadhuryamdaily.com/tour/{t.get('slug', '')}/",
        })
    return items

import streamlit as st


def render_block(block: dict, visit: dict, gh, visit_ref: str) -> dict:
    btype   = block["type"]
    content = block.get("content", {})
    editors = {
        "context_vaishnava":   _edit_context_vaishnava,
        "opening":             _edit_text_block,
        "closing":             _edit_text_block,
        "gurudeva_photos":     _edit_photo_block,
        "sangha_photos":       _edit_photo_block,
        "teaching_atmosphere": _edit_teaching,
        "sangha_portrait":     _edit_sangha_portrait,
        "sangha_voices":       _edit_sangha_voices,
    }
    fn = editors.get(btype)
    if fn:
        return fn(content, visit, gh, visit_ref, btype)
    st.json(content)
    return content


def _edit_text_block(content, visit, gh, visit_ref, btype):
    c1, c2 = st.columns(2)
    pt = c1.text_area("Texto PT", value=content.get("text_pt", ""), height=150, key=btype+"_pt")
    en = c2.text_area("Texto EN", value=content.get("text_en", ""), height=150, key=btype+"_en")
    return {"text_pt": pt, "text_en": en}

def _edit_photo_block(content, visit, gh, visit_ref, btype):

def _edit_sangha_portrait(content, visit, gh, visit_ref, btype):
    ROLES = [
        ("kirtan_leader", "Kirtan",            "Kirtan"),
        ("mridanga",      "Mridanga",           "Mridanga"),
        ("kitchen",       "Cozinha/Prasadam",   "Kitchen/Prasadam"),
        ("pujari",        "Pujari",             "Pujari"),
        ("organizer",     "Organizadores",      "Organizers"),
        ("question",      "Pergunta a Gurudeva","Question to Gurudeva"),
    ]
    performers = gh.get_performers(visit_ref)
    roles      = content.get("roles", [])
    result     = []
    for key, label_pt, label_en in ROLES:
        ex          = next((r for r in roles if r.get("role_pt") == label_pt), {})
        suggestions = [p["name"] for p in performers if p.get("role") == key]
        with st.expander(label_pt):
            if suggestions:
                st.caption("Sugerido: " + ", ".join(suggestions))
            names_str = st.text_input("Nomes (virgula)", value=", ".join(ex.get("names", suggestions)), key="role_"+key)
            note_pt   = st.text_input("Nota PT", value=ex.get("note_pt",""), key="rnote_pt_"+key)

                result.append({"role_pt": label_pt, "role_en": label_en,
                               "names": names, "note_pt": note_pt or None, "note_en": note_en or None})

import streamlit as st
from datetime import datetime, timedelta

def render_days_editor(visits: list) -> list:
    """
    Renderiza editor de dias/visitas.
    Recebe lista de visits e retorna lista atualizada.
    """
    st.subheader("рҹ“… Editor de Visitas")

    if not visits:
        st.info("Nenhuma visita registrada ainda.")
        return visits

    updated_visits = []

    for i, visit in enumerate(visits):
        with st.expander(f"Visita {i+1} вҖ” {visit.get('date', 'sem data')}"):
            col1, col2 = st.columns(2)

            with col1:
                date_val = visit.get("date", str(datetime.today().date()))
                try:
                    date_obj = datetime.strptime(date_val, "%Y-%m-%d").date()
                except:
                    date_obj = datetime.today().date()

                new_date = st.date_input(
                    "Data",
                    value=date_obj,
                    key=f"date_{i}"
                )

            with col2:
                new_title = st.text_input(
                    "TГӯtulo",
                    value=visit.get("title", ""),
                    key=f"title_{i}"
                )

            new_notes = st.text_area(
                "Notas",
                value=visit.get("notes", ""),
                key=f"notes_{i}"
            )

            updated_visits.append({
                **visit,
                "date":  str(new_date),
                "title": new_title,
                "notes": new_notes
            })

    return updated_visits

def _unwrap(response) -> list:
    if isinstance(response, list):
        return response
    if isinstance(response, dict):
        data = response.get("data", response)
        if isinstance(data, dict):
            return data.get("items", [])
        if isinstance(data, list):
            return data
    return []


def _clean(obj):
    if isinstance(obj, dict):
        return {k: _clean(v) for k, v in obj.items() if not k.startswith("_")}
    if isinstance(obj, list):
        return [_clean(i) for i in obj]
    return obj


def build_envelope(payload: dict, parent_tour_key: str) -> dict:
    visit_id = str(payload.get("visit_id", "")).strip()
    if not visit_id:
        raise ValueError("Campo 'visit_id' ausente no JSON.")
    origin_key = f"visit:{visit_id}" if not visit_id.startswith("visit:") else visit_id
    if not parent_tour_key.startswith("tour:"):
        parent_tour_key = f"tour:{parent_tour_key}"
    data = _clean(payload)
    data["schema_version"] = "3.1"
    data["updated_at"] = datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")
    title = payload.get("title_pt") or f"Visita: {visit_id}"
    slug = visit_id.replace(":", "-")
    return {
        "kind":              "visit",
        "origin_key":        origin_key,
        "parent_origin_key": parent_tour_key,
        "title":             title,
        "slug_suggestion":   slug,
        "data":              data,
    }


VALID_STATUSES = {"done", "live", "upcoming", ""}


def validate_payload(payload: dict) -> list:
    errors = []
    for f in ["visit_id", "title_pt", "timezone", "days"]:
        if not payload.get(f):
            errors.append(f"Campo obrigatГіrio ausente: `{f}`")
    days = payload.get("days", [])
    if not isinstance(days, list) or len(days) == 0:
        errors.append("`days` deve ser uma lista nГЈo-vazia.")
        return errors
    if len(days) > 400:
        errors.append(f"`days` excede 400 itens (tem {len(days)}).")
    for i, day in enumerate(days):
        lbl  = day.get("date_local") or f"days[{i}]"
        hero = day.get("hero")
        if not day.get("date_local"):
            errors.append(f"{lbl}: `date_local` ausente.")
        if not day.get("label_pt"):
            errors.append(f"{lbl}: `label_pt` ausente.")
        if not hero:
            errors.append(f"{lbl}: `hero` ausente.")
        else:
            if not hero.get("title_pt"):
                errors.append(f"{lbl} вҖә hero: `title_pt` ausente.")
            if not any(hero.get(k) for k in
                       ["youtube_url", "instagram_url", "facebook_url", "drive_url"]):
                errors.append(f"{lbl} вҖә hero: nenhuma fonte de mГӯdia.")
        for j, s in enumerate(day.get("schedule", [])):
            if not s.get("time_local"):
                errors.append(f"{lbl} вҖә schedule[{j}]: `time_local` ausente.")
            if not s.get("title_pt"):
                errors.append(f"{lbl} вҖә schedule[{j}]: `title_pt` ausente.")
            if s.get("status", "") not in VALID_STATUSES:
                errors.append(f"{lbl} вҖә schedule[{j}]: status invГЎlido `{s.get('status')}`.")
    return errors

@st.cache_data(ttl=120, show_spinner=False)
def _fetch_tours() -> dict:
    FALLBACK = {
        "Tour Espiritual ГҚndia 2026  [tour:india-2026]":                       "tour:india-2026",
        "South America Summer Tour 2025/2026  [tour:south-america-2025-2026]": "tour:south-america-2025-2026",
        "Holand 2026  [tour:holand-2026]":                                     "tour:holand-2026",
    }
    try:
        items = _unwrap(list_tours(per_page=100))
        if not items:
            return FALLBACK
        result = {}
        for t in items:

            ttl = t.get("title") or ok or f"Tour {t.get('id','?')}"
            if ok:
                result[f"{ttl}  [{ok}]"] = ok
        return result if result else FALLBACK
    except Exception:
        return FALLBACK

            p1.markdown(f"**timezone:** `{payload.get('timezone','вҖ”')}`")
            p2.markdown(f"**dias:** `{len(payload.get('days',[]))}`")
            p2.markdown(f"**cover_url:** {'вң…' if payload.get('cover_url') else 'вқҢ vazio'}")
            p2.markdown(f"**tour pai:** `{tour_key}`")
            if payload.get("days"):
                st.dataframe(_day_rows(payload["days"]), use_container_width=True, hide_index=True)

            st.markdown("#### 4 В· ValidaГ§ГЈo")
            errors = validate_payload(payload)
            if errors:
                st.error(f"**{len(errors)} erro(s):**")
                for e in errors:
                    st.markdown(f"- {e}")
                can_send = st.checkbox("вҡ пёҸ Ignorar erros e enviar mesmo assim", key="skip_val")
            else:
                st.success("вң… Schema vГЎlido")
                can_send = True

            if st.checkbox("рҹ”Қ Ver envelope completo", key="show_env"):
                try:
                    ep = build_envelope(payload, tour_key)
                    ep_show = dict(ep)
                    ep_show["data"] = {
                        k: (f"[{len(v)} dias]" if k == "days" else v)
                        for k, v in ep["data"].items()
                    }
                    st.json(ep_show)
                except Exception as e:
                    st.warning(str(e))

            st.divider()
            st.markdown("#### 5 В· Enviar")
            col_btn, col_tip = st.columns([2, 3])
            send = col_btn.button("рҹҡҖ Ingerir visita", type="primary",
                                  disabled=not can_send, key="ingest_send")
            col_tip.caption("Idempotente вҖ” hash idГӘntico вҶ’ `noop` sem writes.")

            if send:
                try:
                    envelope = build_envelope(payload, tour_key)
                    with st.spinner("Enviando..."):
                        result  = ingest_visit(envelope)
                        r_data  = result.get("data", result)
                        action  = r_data.get("action", "?")
                        success = result.get("success", False)

                    LABELS = {
                        "created": ("вң…", "success", "Visita **criada** com sucesso!"),
                        "updated": ("рҹ”„", "success", "Visita **atualizada** com sucesso!"),
                        "noop":    ("рҹ’Ө", "info",    "Sem mudanГ§as вҖ” hash idГӘntico."),
                    }
                    ico, lvl, msg = LABELS.get(action, ("в„№пёҸ", "info", result.get("message", "")))
                    getattr(st, lvl)(f"{ico} {msg}")

                    if success and action != "noop":
                        m1, m2, m3 = st.columns(3)
                        m1.metric("visit_id WP", r_data.get("visit_id", "вҖ”"))
                        m2.metric("AГ§ГЈo", action)
                        m3.metric("tour_id", r_data.get("tour_id", "вҖ”"))
                        h = r_data.get("hash", "")
                        st.caption(f"SHA-256: `{h[:32]}...`" if len(h) > 32 else f"SHA-256: `{h}`")
                        if r_data.get("permalink"):
                            st.link_button("рҹҢҗ Ver visita online", r_data["permalink"])
                        st.cache_data.clear()
                    elif not success:
                        st.error(f"вқҢ {result.get('message','Erro desconhecido')}")
                        with st.expander("рҹ”Қ Resposta completa"):
                            st.json(result)

                except Exception as e:
                    st.error(f"вқҢ Erro: {e}")
